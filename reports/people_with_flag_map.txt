SELECT latlong, ownername || '<br>' || owneraddress
FROM owner o
WHERE o.ExcludeFromBulkEmail = 0 AND o.IsDeceased = 0 AND o.AdditionalFlags LIKE '%$ASK STRING Person flag$%'
ORDER BY o.OwnerSurname
