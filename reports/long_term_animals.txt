Animals who have been on the shelter longer than 3 months.

###
SELECT a.AnimalName, a.ShelterCode, t.AnimalType, i.LocationName,
s.SpeciesName, a.MostRecentEntryDate
FROM animal a
INNER JOIN internallocation i ON i.ID = a.ShelterLocation
INNER JOIN species s ON s.ID = a.SpeciesID
INNER JOIN animaltype t ON t.ID = a.AnimalTypeID
WHERE a.Archived = 0 AND DaysOnShelter > 92
ORDER BY a.MostRecentEntryDate
