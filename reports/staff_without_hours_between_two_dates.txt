All staff who have not worked any hours between two dates
###
en
###
$VAR from DATE Enter from date$
$VAR to DATE Enter to date$

SELECT o.OwnerName
FROM owner o
WHERE o.IsStaff = 1 
AND NOT EXISTS(SELECT ID FROM ownerrota WHERE StartDateTime >= '$@from$' AND EndDateTime <= '$@to$ 23:59:59' AND RotaTypeID < 10)
ORDER BY o.OwnerName
