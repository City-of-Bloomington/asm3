$VAR from DATE Enter from date$
$VAR to DATE Enter to date$
SELECT ReasonName, COUNT(ad.ID) AS Returned
FROM adoption ad
INNER JOIN entryreason ON entryreason.ID = ad.ReturnedReasonID
WHERE ReturnDate >= '$@from$' AND ReturnDate <= '$@to$' 
GROUP BY ReasonName
