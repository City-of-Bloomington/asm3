###

###
SELECT
ShelterCode,
AnimalName,
BreedName,
SizeName,
SexName,
NeuteredName,
AnimalAge,
TimeOnShelter,
IsGoodWithCatsName,
IsGoodWithDogsName,
IsGoodWithChildrenName,
IsHouseTrainedName,
ID AS AnimalID,
DisplayLocation,
AnimalComments,
SpeciesName
FROM v_animal
WHERE Archived = 0 AND IsNotAvailableForAdoption = 0 AND HasTrialAdoption = 0
ORDER BY AnimalName
